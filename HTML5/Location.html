<!DOCTYPE html> <html> <head> <meta charset="utf-8" /> <meta name="viewport" content="width=device-width, initial-scale=1" /> <link rel="stylesheet" href="assets/bootstrap/css/bootstrap.min.css"> <script src="assets/js/jquery-1.8.2.min.js"></script> <title>Current Location Example</title> <script> function showPosition(){ if(navigator.geolocation){ navigator.geolocation.getCurrentPosition(function(position){ var positionInfo = "Your current position is (" + "Latitude: " + position.coords.latitude + ", " + "Longitude: " + position.coords.longitude + ")"; document.getElementById("result").innerHTML = positionInfo; }); } else{ alert("Sorry, your browser does not support HTML5 geolocation."); } } </script> </head> <body> <div id="result" class = "container"> <!--Position information will be inserted here--> </div> <button type="button" class="btn btn-primary" onclick="showPosition();">Current Location</button> </body> </html>
