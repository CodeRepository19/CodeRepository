<!DOCTYPE html> <html> <head> <meta charset="utf-8" /> <meta name="viewport" content="width=device-width, initial-scale=1" /> <link rel="stylesheet" href="../assets/bootstrap/css/bootstrap.min.css"> <title>Drag & Drop Example</title> <style> #div1 { width: 320px; height: 170px; padding: 10px; border: 1px solid #aaaaaa; } </style> <script> function allowDrop(ev) { ev.preventDefault(); } function drag(ev) { ev.dataTransfer.setData("text", ev.target.id); } function drop(ev) { ev.preventDefault(); var data = ev.dataTransfer.getData("text"); ev.target.appendChild(document.getElementById(data)); } </script> </head> <body> <div id="div1" ondrop="drop(event)" ondragover="allowDrop(event)"></div><br> <img id="drag1" src="assets/images/bulb.jpg" draggable="true" ondragstart="drag(event)" width="300" height="140"> </body> </html>
